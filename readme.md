> open-source API for Real-time questions In Hebrew.

## Install

```sh
npm install hebrew-js
```
## import 

```sh

const { Hebrew } = require('hebrew-js')

```
## Usage

```sh

const { 
  direction, 
  meaning, 
  news, 
  translate, 
  weather 
} = new Hebrew()

direction('转 ','专砖')
news('拽专')
translate('转专 转 驻拽爪 注专转')
weather('驻转 转拽')
meaning('驻住驻')

```
